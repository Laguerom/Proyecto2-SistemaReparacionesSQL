/*--Se crea la base de datos*/

USE Prueba6;



-- CREACIÓN DE TABLAS --

-- Tabla de Usuarios --
CREATE TABLE Usuarios(
	UsuarioID INT IDENTITY(1,1) PRIMARY KEY,
	CedulaID VARCHAR(50) UNIQUE NOT NULL,
	Nombre VARCHAR(100) NOT NULL,
	CorreoElectronico VARCHAR(100) UNIQUE NOT NULL,
	Telefono VARCHAR(20)
	);

-- Tabla de Equipos --
CREATE TABLE Equipos(
	EquipoID INT IDENTITY(1,1) PRIMARY KEY,
	Modelo VARCHAR(100) NOT NULL,
	Marca VARCHAR(100),
	UsuarioID INT,
	FOREIGN KEY (UsuarioID) REFERENCES Usuarios(UsuarioID)
	);

-- Tabla de tecnicos --
CREATE TABLE Tecnicos(
	TecnicoID INT IDENTITY(1,1) PRIMARY KEY,
	Nombre VARCHAR(100) NOT NULL,
	Especialidad VARCHAR(100)
	);

-- Tabla de Reparaciones --
CREATE TABLE Reparaciones(
	ReparacionID INT IDENTITY(1,1) PRIMARY KEY,
	EquipoID INT,
	FechaSolicitud DATETIME,
	Estado VARCHAR(50) DEFAULT 'Pendiente',
	FOREIGN KEY (EquipoID) REFERENCES Equipos(EquipoID)
	);

-- Tabla de Asignaciones --
CREATE TABLE Asignaciones(
	AsignacionID INT IDENTITY(1,1) PRIMARY KEY,
	ReparacionID INT,
	TecnicoID INT,
	FOREIGN KEY (ReparacionID) REFERENCES Reparaciones(ReparacionID),
	FOREIGN KEY (TecnicoID) REFERENCES Tecnicos(TecnicoID)
	);

-- Tabla de DetallesReparacion --
CREATE TABLE DetallesReparacion(
	DetalleID INT IDENTITY(1,1) PRIMARY KEY,
	ReparacionID INT,
	Descripcion VARCHAR(255) DEFAULT '_' ,
	FechaFinalizacion DATETIME,
	FOREIGN KEY (ReparacionID) REFERENCES Reparaciones(ReparacionID)
	);

--Consulta Extra
/*Esta consulta busca a un usuario usando su CedulaID, columna agregada a la tabla Usuarios. 
Luego muestra los equipos que tiene registrados,
las reparaciones que se han realizado a esos equipos y los detalles de cada reparación. 

Esto se logra usando los comandos:

	SELECT: Seleccionando la informacion que se busque 
	mostrar de cada tabla

	INNER JOIN: Une las tablas mostrando solo los datos 
	que coinciden entre ellas.

	LEFT JOIN: Permite mostrar todas las reparaciones 
	aunque no tengan todavía un detalle registrado.

	WHERE: Filtra la CedulaID para encontrar al usuario 
	específico que se desea consultar.*/


SELECT 
    Usuarios.CedulaID,
    Usuarios.Nombre,
    Equipos.EquipoID,
    Equipos.Modelo,
    Equipos.Marca,
    Reparaciones.ReparacionID,
    Reparaciones.FechaSolicitud,
    Reparaciones.Estado,
    DetallesReparacion.DetalleID,
    DetallesReparacion.Descripcion,
    DetallesReparacion.FechaFinalizacion
FROM Usuarios
INNER JOIN Equipos
    ON Usuarios.UsuarioID = Equipos.UsuarioID
INNER JOIN Reparaciones
    ON Equipos.EquipoID = Reparaciones.EquipoID
LEFT JOIN DetallesReparacion
    ON Reparaciones.ReparacionID = DetallesReparacion.ReparacionID
WHERE Usuarios.CedulaID = '145094515';
